How to set up Google Sheets sync
What this does Every saved part is automatically appended as a new row in your Google Sheet, including the GPS coordinates and a link to Google Maps. Photos are stored locally (too large for Sheets).
Go to sheets.google.com and create a new spreadsheet. Copy its URL — you'll need it later.
Open Extensions → Apps Script in your Google Sheet.
Replace all existing code with the script below, then click Save.
// PieceTrack — Google Apps Script // Paste this into your Google Apps Script editor // PieceTrack — Google Apps Script // Paste this into your Google Apps Script editor const SHEET_NAME = 'Parts'; const HEADERS = ['ID', 'Date', 'Piece ID', 'Package No.', 'Comment', 'Latitude', 'Longitude', 'GPS Source', 'Maps Link', 'Synced At']; function doPost(e) { try { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow(HEADERS); sheet.getRange(1, 1, 1, HEADERS.length).setFontWeight('bold').setBackground('#F6C90E'); sheet.setFrozenRows(1); } const rows = Array.isArray(data) ? data : [data]; rows.forEach(p => { sheet.appendRow([ p.id, p.date, p.piece, p.package, p.comment || '', p.lat ?? '', p.lng ?? '', p.gpsSource || '', p.lat ? `http://maps.google.com/maps?q=${p.lat},${p.lng}` : '', new Date().toISOString() ]); }); return ContentService .createTextOutput(JSON.stringify({status:'ok', count:rows.length})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({status:'error', message:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { return ContentService .createTextOutput(JSON.stringify({status:'ok', rows:[]})) .setMimeType(ContentService.MimeType.JSON); } const [headers,...rows] = sheet.getDataRange().getValues(); const parts = rows.map(r => Object.fromEntries(headers.map((h,i)=>[h,r[i]]))); const piece = e.parameter.piece || ''; const pkg = e.parameter.package || ''; const filtered = parts.filter(p => (!piece || String(p['Piece ID']).toLowerCase().includes(piece.toLowerCase())) && (!pkg || String(p['Package No.']).toLowerCase().includes(pkg.toLowerCase())) ); return ContentService .createTextOutput(JSON.stringify({status:'ok', rows:filtered})) .setMimeType(ContentService.MimeType.JSON); }
Click Deploy → New deployment. Set type to Web App. Set Execute as to Me and Who has access to Anyone. Click Deploy and authorise.
Copy the Web App URL that appears (starts with https://script.google.com/macros/s/…) and paste it in the field above.
Click Test connection. If successful, all future parts will sync automatically.
