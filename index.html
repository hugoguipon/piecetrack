Google Drive — Photo Storage
Photos stored locally only Same Apps Script URL handles Drive uploads — no extra config needed.
How Drive upload works When you save a part, the compressed photo is uploaded to a PieceTrack/ folder in the Google Drive of your Apps Script account. A direct link is then written into the Google Sheet row. Photos stored locally on the device are kept as backup.
Photo compression
Current target: Balanced (~300 KB) Compression uses iterative quality reduction to hit the target size. The original photo is never modified — only the compressed version is uploaded and stored.

// PieceTrack v2 — Google Apps Script // Handles: Sheets rows + Google Drive photo upload const SHEET_NAME = 'Parts'; const DRIVE_FOLDER = 'PieceTrack'; const HEADERS = ['ID', 'Date', 'Piece ID', 'Package No.', 'Comment', 'Latitude', 'Longitude', 'GPS Source', 'Maps Link', 'Photo URL', 'Photo Size KB', 'Synced At']; function getOrCreateFolder() { const folders = DriveApp.getFoldersByName(DRIVE_FOLDER); return folders.hasNext() ? folders.next() : DriveApp.createFolder(DRIVE_FOLDER); } function doPost(e) { try { const data = JSON.parse(e.postData.contents); const rows = Array.isArray(data) ? data : [data]; const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow(HEADERS); sheet.getRange(1, 1, 1, HEADERS.length) .setFontWeight('bold') .setBackground('#F6C90E'); sheet.setFrozenRows(1); } const folder = getOrCreateFolder(); const results = []; rows.forEach(p => { let photoUrl = ''; let photoSizeKB = ''; // Upload photo to Drive if provided if (p.photoB64) { try { const decoded = Utilities.base64Decode(p.photoB64); const blob = Utilities.newBlob(decoded, 'image/jpeg', `${p.piece}_${p.package}_${p.id}.jpg`); const file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); photoUrl = `https://drive.google.com/uc?id=${file.getId()}`; photoSizeKB = Math.round(decoded.length / 1024); } catch(driveErr) { photoUrl = 'upload-failed'; } } sheet.appendRow([ p.id, p.date, p.piece, p.package, p.comment || '', p.lat ?? '', p.lng ?? '', p.gpsSource || '', p.lat ? `https://maps.google.com/maps?q=${p.lat},${p.lng}` : '', photoUrl, photoSizeKB, new Date().toISOString() ]); results.push({ id: p.id, photoUrl }); }); return ContentService .createTextOutput(JSON.stringify({ status: 'ok', count: rows.length, results })) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ status: 'error', message: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) return ContentService .createTextOutput(JSON.stringify({ status: 'ok', rows: [] })) .setMimeType(ContentService.MimeType.JSON); const [headers, ...rows] = sheet.getDataRange().getValues(); const parts = rows.map(r => Object.fromEntries(headers.map((h, i) => [h, r[i]]))); const piece = e.parameter.piece || ''; const pkg = e.parameter.package || ''; const filtered = parts.filter(p => (!piece || String(p['Piece ID']).toLowerCase().includes(piece.toLowerCase())) && (!pkg || String(p['Package No.']).toLowerCase().includes(pkg.toLowerCase())) ); return ContentService .createTextOutput(JSON.stringify({ status: 'ok', rows: filtered })) .setMimeType(ContentService.MimeType.JSON); }
